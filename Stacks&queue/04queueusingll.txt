class MyQueue {
    QueueNode front, rear;

    void push(int a) {
        QueueNode temp = new QueueNode(a);
        if (rear == null) {
            front = rear = temp;
            return;
        }
        rear.next = temp;
        rear = temp;
    }

    int pop() {
        if (front == null)
            return -1;
        int val = front.data;
        front = front.next;
        if (front == null)
            rear = null;
        return val;
    }
}
