class Solution {
    public int[] maxSlidingWindow(int[] nums, int k) {
         int n= nums.length;
         int [] ans= new int[n-k+1];
         int ansIndex=0;
         Deque<Integer> dq= new LinkedList<>();
         for(int i =0;i<n;i++){
            while(!dq.isEmpty() && dq.peekFirst()<=i-k ){
                dq.pollFirst();
            }
            while(!dq.isEmpty() && nums[dq.peekLast()]<=nums[i]){
                dq.pollLast();
            }
            dq.offerLast(i);
            if(i>=k-1){
                int max=nums[dq.peekFirst()];
                ans[ansIndex]=max;
                ansIndex++;
            }
         }
         return ans;
         

    }
}
