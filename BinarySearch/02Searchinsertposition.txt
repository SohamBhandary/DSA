class Solution {
    public int searchInsert(int[] nums, int target) {
        int n = nums.length;
        int l=0;
        int h=n-1;
        int ans=n;
        while(l<=h){
            int mid=(l+h)/2;
            if(target<=nums[mid]){
                ans=mid;
                h=mid-1;
            }else{
                l=mid+1;
            }
        }
        return ans;
        
    }
}
