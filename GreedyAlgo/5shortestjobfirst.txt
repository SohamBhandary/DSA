class Solution {
    static int solve(int bt[]) {
        Arrays.sort(bt);
        long waitTime=0;
        long totalTime=0;
        int n = bt.length;
        for(int i=0;i<n;i++){
            waitTime=waitTime+totalTime;
            totalTime=totalTime+bt[i];
        }
        return (int)(waitTime/n);
       
        
    }
}
